def to_s
  majorkeys = ['C flat', 'G flat', 'D flat', 'A flat', 'E flat', 'B flat', 'F',
    'C', 'G', 'D', 'A', 'E', 'B', 'F#', 'C#']
  minorkeys = ['a flat', 'e flat', 'b flat', 'f', 'c', 'g', 'd',
    'a', 'e', 'b', 'f#', 'c#', 'g#', 'd#', 'a#']
  minor_key? ? "key sig #{minorkeys[sharpflat + 7]} minor" :
    "key sig #{majorkeys[sharpflat + 7]} major"
end
